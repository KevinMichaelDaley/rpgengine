# Physics Tick Time Budget (Current)
# ================================
# Target: ~1.5–2.0 ms at 30 Hz (typical: 2 substeps)
# Percentages are of total physics tick CPU time (not wall-clock).
#
# NOTE:
# - Halo closure and broadphase are executed once per tick (outside the substep loop).
# - Buffer swap happens once per substep.
# - XPBD + explicit position-projection stages are not currently part of tick.c; split impulse
#   position correction is fused into TGS via pseudo-velocities.
#
# Zone names follow pattern: Phys.Stage.DescriptiveParticiple
#
# Format: FUNCTION_NAME                          TRACY_ZONE                              %TIME
# ----------------------------------------------------------------------------------------------

# TICK PROLOGUE
phys_frame_arena_reset                           Phys.Tick.ResettingArena                 0.5

# STAGE 0: STEP PLAN
phys_stage_step_plan                             Phys.Plan.ComputingStepPlan              0.5

# STAGE 1: TIER CLASSIFICATION
phys_stage_tier_classify                         Phys.Tier.ClassifyingBodies              3.0

# STAGE 2: SPATIAL UPDATE (AABBs + grid build)
phys_stage_spatial_update                        Phys.Spatial.UpdatingGrid                3.0

# STAGE 3: HALO CLOSURE (once per tick)
phys_stage_halo_closure                          Phys.Halo.ClosingNeighbors               4.0

# STAGE 5: BROADPHASE (once per tick)
phys_stage_broadphase                            Phys.Broad.FindingPairs                  8.0

# --- SUBSTEP LOOP (typically 2 substeps) ---

# STAGE 4: AABB UPDATE (substep>0)
phys_stage_aabb_update                           Phys.AABB.UpdatingBounds                 3.0

# STAGE 6: NARROWPHASE
phys_stage_narrowphase                           Phys.Narrow.TestingCollisions           14.0

# STAGE 7: MANIFOLD BUILD
phys_stage_manifold_build                        Phys.Manifold.BuildingContacts           5.0

# STAGE 8: STABILIZATION
phys_stage_stabilization                         Phys.Stab.ComputingHints                 4.0

# STAGE 9: CONSTRAINT BUILD (includes joint constraint build)
phys_stage_constraint_build                      Phys.Constraint.BuildingJacobians        5.0
# Joint rows: phys_joint_build_ball/distance/hinge -> phys_joint_build_constraints

# STAGE 10: ISLAND BUILD (T0/T1)
phys_stage_island_build                          Phys.Island.BuildingComponents           5.0

# STAGE 10b: SUB-SUBSTEPS (fast islands only, before main TGS)
# Islands with max body speed > 15 m/s get 1–8 sub-substeps.
# Each sub-substep: rebuild joint constraints, TGS solve, inline integrate.
compute_island_sub_substeps                      Phys.SubSub.ComputingCount               0.5
rebuild_island_joint_constraints                 Phys.SubSub.RebuildingJoints             1.0
# (sub-substep TGS + integrate time varies with island speed)

# STAGE 11: TGS SOLVE (includes split impulse + nonlinear joint projection)
# Adaptive iterations: 1×–5× base (8 iters), sqrt ramp based on max island speed.
# Large-island caps: ≤256 constraints → 5×, 257-512 → 3×, >512 → 2×.
# SOR omega=1.1 (successive over-relaxation for faster convergence).
# Warmstarting: lambda writeback to manifold cache for initial guess.
# Speed-dependent Baumgarte leak for joints (0–0.6 over 5–60 m/s).
# Per-row viscous damping for joint constraints (damping < 1).
# After iterations: 4 nonlinear joint projection passes.
phys_stage_tgs_solve                             Phys.Solve.IteratingTGS                 30.0
project_joints_nonlinear                         Phys.Solve.ProjectingJoints              2.0

# STAGE 12: INTEGRATE (sub-substepped bodies are skipped)
phys_stage_integrate                             Phys.Integrate.UpdatingPositions         5.0

# STAGE 13: CACHE COMMIT
phys_stage_cache_commit                          Phys.Cache.CommittingImpulses            2.0

# BUFFER SWAP (per substep)
phys_body_pool_swap_buffers                      Phys.Tick.SwappingBuffers                0.5

# EPILOGUE
phys_manifold_cache_expire                       Phys.Cache.ExpiringOld                   0.5

# NETWORK (outside physics tick; budget shown here for convenience)
# This is not part of `phys_world_tick*()`, but commonly paired with it in a server tick.
phys_snapshot_encode                             Phys.Net.EncodingSnapshot                3.0
phys_snapshot_decode                             Phys.Net.DecodingSnapshot                2.0
phys_snapshot_encode_delta                       Phys.Net.EncodingDelta                   2.0
phys_snapshot_decode_delta                       Phys.Net.DecodingDelta                   1.5
